my.sample <- function(x, size, prob = NULL){
  # x: vector
  # 放回抽样
  l <- length(x)
  
  inversefunc <- function(u){
    return(x[ceiling(u*l)])
  }
  
  u <- runif(size)
  x <- inversefunc(u)
  
  return(x)
}

Ex 3.2

standard Laplace distribution, pdf: \(f(x) = \frac{1}{2}e^{-|x|} = \begin{cases} \frac{1}{2}e^{x}, & x \le 0 , \\ \frac{1}{2}e^{-x}, & x \gt 0, \end{cases}\) cdf: \(F(x) = \begin{cases} \frac{1}{2}e^{x}, & x \le 0 , \\ 1-\frac{1}{2}e^{-x}, & x \gt 0, \end{cases}\)

\(F^{-1}(x) = \begin{cases} \log(2x), & 0 \lt x \le \frac{1}{2} , \\ -\log(2-2x), &\frac{1}{2} \lt x \lt 1, \end{cases}\)

u <- runif(1000)
x <- rep(0, 1000)
Laplace <- function(x){
  y=x
  for (i in 1:length(x)) {
    if(x[i]<0) y[i]=1/2*exp(x[i])
    else y[i]=1/2*exp(-x[i])
  }
  
  return(y)
}
for (i in 1:1000) {
  if(u[i] < 1/2)
    x[i] = log(2*u[i])
  else
    x[i] = -log(2-2*u[i])
}
hist(x, prob = TRUE)
y <- seq(-6, 6, 0.01)
lines(y, Laplace(y))

Ex 3.7

\(\rm{Beta}(a, b)\) distribution, pdf: \(f(x)=\frac {\Gamma (\alpha +\beta )}{\Gamma (\alpha )\Gamma (\beta )}x^{\alpha -1}(1-x)^{\beta -1}\)

my.beta <- function(a, b, n){
  j=k=0
  y <- numeric(n)
  while (k < n) {
    u <- runif(1)
    j <- j + 1
    x <- runif(1) 
    if (x^(a-1) * (1-x)^(b-1) > u) {
      k <- k + 1
      y[k] <- x
    }
  }
return(y)
}
y <- my.beta(3,2,1000)
hist(y, prob = T)

my.epan <- function(n){
  x <- rep(0, n)
  for (i in 1:n) {
    u1 <- runif(1, -1, 1)
    u2 <- runif(1, -1, 1)
    u3 <- runif(1, -1, 1)
    if(abs(u3)>=abs(u2)&&abs(u3)>=abs(u1)) x[i] <- u2
    else x[i] <- u3
  }
  return(x)
}
x <- my.epan(1000)
hist(x, prob = TRUE)

Ex 3.10

Proof:

\((U_1, U_2, U_3)\)服从\([-1,1]\times[-1,1]\times[-1,1]\)上的均匀分布, \(P(X\le x)=P(U_2\le x||U_3|\ge|U_2| and |U_3|\ge|U_1|) P(|U_3|\ge|U_2| and |U_3|\ge|U_1|) + P(U_3\le x||U_3|\lt|U_2| or |U_3|\lt|U_1|) P(|U_3|\ge|U_2| and |U_3|\ge|U_1|)\) 其中\(P(|U_3|\ge|U_2| and |U_3|\ge|U_1|)=P(|U_3|=max\{|U_1|,|U_2|,|U_3|\})=\frac{1}{3}\)

\(P(U_3\le x||U_3|\lt|U_2| or |U_3|\lt|U_1|)=\frac{2}{3}\)

\(\{|U_3|\ge|U_2| and |U_3|\ge|U_1|\}\)在\((U_1, U_2, U_3)\)分布范围内的体积为\(\frac{1}{3}\times2\times2\times2=\frac{8}{3}\), \(\{|U_3|\lt|U_2| or |U_3|\lt|U_1|\}\) 的对应体积为\(2^3-\frac{8}{3}=\frac{16}{3}\).

\(P(U_2\le x||U_3|\ge|U_2| and |U_3|\ge|U_1|) = 1-\frac{\frac{5}{6}x^3-\frac{3}{2}x^2-\frac{3}{2}x+\frac{5}{6}}{8/3}\)

\(P(U_3\le x||U_3|\lt|U_2| or |U_3|\lt|U_1|) = 1-\frac{(1-x)^2+\frac{1}{3}x^3-x+\frac{2}{3}}{16/3}\)

cdf: \(F(x)=P(X\le x) = \frac{1}{3} \times (1-\frac{\frac{5}{6}x^3-\frac{3}{2}x^2-\frac{3}{2}x+\frac{5}{6}}{8/3}) + \frac{2}{3} \times 1-\frac{(1-x)^2+\frac{1}{3}x^3-x+\frac{2}{3}}{16/3}\)

pdf: \(f(x) = F'(x) = \frac{3}{4}(1-x^2)\) 即为Epanechnikov kernel.
